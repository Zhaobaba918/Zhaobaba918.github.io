35 二分查找

二分查找只有一种特殊情况，就是 start>end

剩下按部就班写就行

  1. 非递归
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int start = 0,end = nums.size()-1,med;

        while(start<=end){
            med = (start+end)/2;
            if (target == nums[med]){
                return med;
            }else if (target < nums[med]){
                end = med -1;
            }else {
                start = med+1;
            }
        }
        return start;
    }
};
  1. 递归
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        return binsearch(nums,0,nums.size()-1,target);
    }

    int binsearch(vector<int>& nums,int start,int end,int target){
        int med = (start+end)/2;
        if (start>end){
            return start;
        }
        else if (target == nums[med]){
            return med;
        }
        else if (target < nums[med]){
            return binsearch(nums,start,med-1,target);
        }
        else{
            return binsearch(nums,med+1,end,target);
        }
    }
};

209 长度最小的子数组

滑动窗口法O（n）

class Solution {
public:
    int minSubArrayLen(int target, vector<int>& nums) {
        int beg = 0,end = 0,sum = nums[0], min = nums.size()+1;
        while (1){
            if (sum < target){
                if (++end >= nums.size()){
                    break;
                }
                sum += nums[end];
            }
            else{
                if (end-beg+1<min){
                    min = end-beg+1;
                    if (min == 1){
                        return 1;
                    }
                }
                sum -= nums[beg++];
            }
        }
        if (min <= nums.size()){
            return min;
        }else{
            return 0;
        }
    }
};

nlogn

一般是一个O(n)复合一个logn

比如一个遍历和一个二分查找

可以是O(n^2)优化出来的

997 找到小镇的法官（图，统计出度入度）

class Solution {
public:
    int findJudge(int n, vector<vector<int>>& trust) {
        vector<bool> notJudge(n+1,false);
        vector<int> beTrusted(n+1);
        int i=0;
        for (;i<trust.size();i++){
            notJudge[trust[i][0]] = 1;
            beTrusted[trust[i][1]]++;
        }
        for (i=1;i<=n;i++){
            if (notJudge[i]==0 && beTrusted[i]>=n-1){
                return i;
            }
        }
        return -1;
    }
};

可以进一步优化为只用一个数组，一个入度+1，一个出度-1，最后必须是n-1才是法官
